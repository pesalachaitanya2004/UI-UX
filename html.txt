<script>
  // Live clock
  function updateClock() {
    const now = new Date();
    const date = now.toLocaleDateString('en-GB', { weekday:'short', day:'numeric', month:'short' });
    const time = now.toLocaleTimeString('en-GB', { hour:'2-digit', minute:'2-digit' });
    document.getElementById('clock').textContent = `${date} — ${time}`;
  }
  updateClock();
  setInterval(updateClock, 1000);

  // Sidebar nav
  function setNav(el) {
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
    el.classList.add('active');
  }

  // Room select
  function selectRoom(el) {
    document.querySelectorAll('.room-card').forEach(r => r.classList.remove('selected'));
    el.classList.add('selected');
  }

  // Device toggle
  function toggleDevice(input) {
    const row = input.closest('.device-row');
    row.classList.toggle('on', input.checked);
    const status = row.querySelector('.device-status');
    if (!input.checked) {
      status.dataset.prev = status.textContent;
      status.textContent = 'Off';
    } else {
      status.textContent = status.dataset.prev || 'On';
    }
  }

  // Brightness slider
  function updateBrightness(val) {
    document.getElementById('bval').textContent = val + '%';
  }

  // Thermostat
  let temp = 22;
  function changTemp(delta) {
    temp = Math.max(16, Math.min(30, temp + delta));
    document.getElementById('thermoTemp').textContent = temp + '°';
    const pct = (temp - 16) / (30 - 16);
    const offset = 465 - pct * 465 * 0.75;
    document.getElementById('ringFill').style.strokeDashoffset = offset;
  }

  // Mode buttons
  function setMode(btn) {
    document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
  }

  // Scene buttons
  function setScene(btn) {
    document.querySelectorAll('.scene-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
  }

  // Energy bar chart
  const hours = ['6a','7','8','9','10','11','12p','1','2','3','4','5','6','7','8'];
  const usage = [1.2,1.8,2.1,2.4,3.0,3.6,4.1,3.8,3.4,3.2,3.8,4.2,4.5,4.2,3.9];
  const maxU = Math.max(...usage);
  const chart = document.getElementById('energyChart');

  hours.forEach((h, i) => {
    const pct = usage[i] / maxU;
    const color = pct > 0.85 ? 'var(--red)' : pct > 0.65 ? 'var(--accent3)' : 'var(--accent)';
    const wrap = document.createElement('div');
    wrap.className = 'bar-wrap';
    wrap.innerHTML = `
      <div class="bar" style="height:${Math.round(pct*88)}px;background:${color};opacity:0.8"></div>
      <div class="bar-label">${h}</div>
    `;
    chart.appendChild(wrap);
  });
</script>
